def load_sffsd_data(file_path='./data/S-FFSD.csv'):
    """S-FFSD"""
    data = pd.read_csv(file_path)
    labeled_data = data[data['Labels'].isin([0, 1])].copy()

    X = labeled_data.drop('Labels', axis=1)
    y = labeled_data['Labels']

    categorical_features = ['Source', 'Target', 'Location', 'Type']
    for col in categorical_features:
        le = LabelEncoder()
        X[col] = le.fit_transform(X[col])

    numerical_features = ['Time', 'Amount']
    scaler = StandardScaler()
    X[numerical_features] = scaler.fit_transform(X[numerical_features])

    return train_test_split(
        X.values, y.values,
        test_size=0.4,
        random_state=42,
        stratify=y
    )

def evaluate_model(model, X_test, y_test, dataset_name):
    import matplotlib.pyplot as plt
    import seaborn as sns
    from sklearn.metrics import (
        accuracy_score, precision_score, recall_score,
        f1_score, roc_auc_score, confusion_matrix,
        average_precision_score, precision_recall_curve
    )
    import platform
    import os
    plt.rcParams.update({'font.size': 12})  # 

    # 
    font_candidates = [
        'Microsoft YaHei',   # Windows
        'SimHei',            # Windows
        'PingFang SC',       # Mac
        'Source Han Sans SC',# 
        'WenQuanYi Zen Hei', # Linux
        'Arial Unicode MS'   # 
    ]
    current_font = None
    for font in font_candidates:
        try:
            plt.rcParams['font.sans-serif'] = font
            plt.rcParams['axes.unicode_minus'] = False
            # 
            plt.figure()
            plt.text(0.5, 0.5, '', fontfamily=font)
            plt.close()
            current_font = font
            break
        except:
            continue

    if current_font is None:
        font_paths = [
            'C:/Windows/Fonts/msyh.ttc',  # Windows 
            '/System/Library/Fonts/PingFang.ttc',  # Mac
            '/usr/share/fonts/wenquanyi/wqy-zenhei.ttc'  # Linux
        ]
        for path in font_paths:
            if os.path.exists(path):
                from matplotlib.font_manager import FontProperties
                font_prop = FontProperties(fname=path)
                plt.rcParams['font.family'] = font_prop.get_name()
                current_font = font_prop.get_name()
                break

    print(f": {current_font or ''}")

    y_pred = model.predict(X_test)
    y_proba = model.predict_proba(X_test)[:, 1]  # 

    metrics = {
        '': accuracy_score(y_test, y_pred),
        '': precision_score(y_test, y_pred, zero_division=0),
        '': recall_score(y_test, y_pred, zero_division=0),
        'F1': f1_score(y_test, y_pred, zero_division=0),
        'AUC': roc_auc_score(y_test, y_proba),
        'AP': average_precision_score(y_test, y_proba)  # 
    }

    print(f"\n {dataset_name} :")
    print("="*40)
    for name, value in metrics.items():
        print(f"{name:<6}: {value:.4f}")
    print("="*40)

    plt.figure(figsize=(8, 6))
    cm = confusion_matrix(y_test, y_pred)
    ax = sns.heatmap(cm, annot=True, fmt='d', cmap='Blues',
                    xticklabels=['', ''],
                    yticklabels=['', ''],
                    annot_kws={'fontsize': 12})

    def set_font(ax, font):
        for item in ([ax.title, ax.xaxis.label, ax.yaxis.label] +
                    ax.get_xticklabels() + ax.get_yticklabels()):
            item.set_fontproperties(plt.matplotlib.font_manager.FontProperties(family=font))

    set_font(ax, current_font)
    plt.title(f'{dataset_name} - -', pad=20)
    plt.xlabel('')
    plt.ylabel('')
    plt.tight_layout()
    plt.show()

    # 6.2 PR
    precision, recall, _ = precision_recall_curve(y_test, y_proba)
    plt.figure(figsize=(8, 6))
    plt.plot(recall, precision, lw=2, color='#FF7043')
    plt.fill_between(recall, precision, alpha=0.2, color='#FFAB91')

    ax = plt.gca()
    set_font(ax, current_font)
    plt.title(f'{dataset_name} - PR- (AP={metrics["AP"]:.3f})', pad=20)
    plt.xlabel('')
    plt.ylabel('')
    plt.xlim([0.0, 1.0])
    plt.ylim([0.0, 1.05])
    plt.grid(True, linestyle='--', alpha=0.3)
    plt.tight_layout()
    plt.show()

    return metrics
